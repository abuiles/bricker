##
# Just a plain old show on a node. Does not return any edges
# but just properties.

GET /node/15

Response: 200 OK

Body:
    { id: "a3efdd48c8329"
    , type: "Movie"                       # Contains the node type
    , payload: { title: "Matrix"
               , production_year: 1999 }
    }
