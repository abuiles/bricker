##
# Just a plain old show on a node. Does not return any edges
# but just properties.

GET /node/15

Response: 200 OK

Body:
    { id: "a3efdd48c8329"
    , type: "Movie"                       # Contains the node type
    , payload: { title: "Matrix"
               , production_year: 1999 }
    }

##
# Getting a node of the type "Person" 

GET /node/16

Response: 200 OK

Body:
    { id: "a93f8d83c02b38"
    , type: "Person"
    , payload: { name: "Johnny Depp"
               , sex: "Male" }
    }

##
# Getting a non-existant node will result in this error

GET /node/99999

Response: 404 Not Found

##
# Node properties can be updated by issuing a PUT request
# to its url. Note that all properties will be deleted
# and replaced with the ones you supplied. That means you 
# always overwrite everything.

PUT /nodes/15

Body:
    { "somekey": "someval" }

Response: 200 OK

Body:
    { "somekey": "someval" }
